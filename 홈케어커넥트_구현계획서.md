```
│ 항목          │ PWA      │ Flutter  │ RN+Expo  │ 네이티브(별도개발)  │

│ 코드 공유     │ Next.js  │ Dart만   │ TS 공유  │ 불가               │

│ 푸시 알림     │ △ Web   │ ◎ FCM   │ ◎ FCM   │ ◎ APNs/FCM        │
│ 개발 속도     │ 빠름     │ 보통    │ 빠름     │ 느림(2배)          │
│ 생태계        │ 풍부     │ 성장중  │ 매우풍부 │ 플랫폼별           │
│ 팀 역량 활용  │ ◎ TS    │ △ Dart │ ◎ TS    │ △ Swift/Kotlin     │
├──────────────┴──────────┴──────────┴──────────┴────────────────────┤
│ 최종 선택: React Native + Expo                                      │
│                                                                     │

│ ① TypeScript 단일 언어로 웹+앱+서버 전체 개발 → 팀 효율 극대화      │
│ ② Expo EAS Build/Submit → 앱스토어 빌드·제출 자동화                 │
│ ③ 네이티브 GPS, 카메라, 푸시알림 완벽 지원 → 간호사앱 핵심 기능     │

│ ⑤ OTA 업데이트 (EAS Update) → 심사 없이 즉시 버그 수정 가능        │
│ ⑥ Apple/Google 심사 통과율 높음 (네이티브 렌더링)                   │
└─────────────────────────────────────────────────────────────────────┘
```

#### 왜 Next.js 15인가? (웹)

```
병원 관리 콘솔 + 플랫폼 백오피스 = PC 웹 전용
  → 앱스토어 심사 불필요 → 웹 프레임워크 최적

Next.js 15 선택 이유:
  ① App Router + React Server Components → 초기 로딩 속도
  ② Server Actions → Supabase 서버사이드 호출 (API Key 보호)
  ③ Vercel 배포 → 제로 설정, 글로벌 CDN, 자동 스케일링
  ④ React 19 → use() hook, 서버/클라이언트 최적 분리
  ⑤ 한국 개발자 풀 최대 → 채용 유리
```

#### 왜 TypeScript 단일 언어인가?

```
모노레포 내 모든 프로젝트가 TypeScript를 사용하면:

  ① 공유 패키지: 타입 정의, 유틸리티, API 클라이언트, 유효성 검증 로직을
     웹과 모바일이 동일하게 사용
  ② Supabase 타입 자동 생성: supabase gen types → DB 스키마 타입 자동 동기화
  ③ 개발자 4명 전원이 웹/앱/서버 교차 작업 가능
  ④ 코드 리뷰, 디버깅, 리팩토링 비용 최소화
```

```sql
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.profiles (id, role, full_name, phone)
  VALUES (
    NEW.id,
    COALESCE(NEW.raw_user_meta_data->>'role', 'guardian'),
    COALESCE(NEW.raw_user_meta_data->>'full_name', ''),
    COALESCE(NEW.raw_user_meta_data->>'phone', '')
  );
  RETURN NEW;
END;
```

```sql
  ON public.profiles FOR SELECT
  USING (auth.uid() = id);

  ON public.profiles FOR UPDATE
  USING (auth.uid() = id);

-- patients 정책
-- =====================================================
-- 보호자: 자신이 연결된 환자만 조회
CREATE POLICY "Guardians can view linked patients"
  ON public.patients FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM public.guardian_patient_links
      WHERE patient_id = patients.id AND guardian_id = auth.uid()
    )
  );

-- 간호사: 자신이 담당하는 환자만 조회
CREATE POLICY "Nurses can view assigned patients"
  ON public.patients FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM public.visits v
      JOIN public.staff s ON s.id = v.nurse_id
      WHERE v.patient_id = patients.id
        AND s.user_id = auth.uid()
        AND v.status NOT IN ('cancelled')

  );

CREATE POLICY "Org admins can view org patients"
  ON public.patients FOR SELECT
  USING (

      SELECT 1 FROM public.service_plans sp
```

### 3.4 핵심 데이터베이스 함수

```sql
-- =====================================================
-- AI 매칭: 반경 내 기관 검색 + 점수 산출
-- =====================================================
CREATE OR REPLACE FUNCTION public.find_matching_organizations(
  p_patient_id UUID,
  p_radius_km NUMERIC DEFAULT 10
)
RETURNS TABLE (
  org_id UUID,
  org_name TEXT,
  distance_km NUMERIC,
  travel_minutes NUMERIC,
  service_match_score NUMERIC,
  capacity_score NUMERIC,
  reputation_score NUMERIC,
  response_score NUMERIC,
  total_score NUMERIC
) AS $$
DECLARE
  v_patient_location GEOGRAPHY;
  v_needed_services TEXT[];
BEGIN
  -- 환자 위치/필요 서비스 조회
  SELECT location, needed_services
    INTO v_patient_location, v_needed_services
    FROM public.patients WHERE id = p_patient_id;

  RETURN QUERY
  SELECT
    o.id AS org_id,
    o.name AS org_name,

    -- 거리 (km)
    ROUND((ST_Distance(o.location, v_patient_location) / 1000)::NUMERIC, 1) AS distance_km,

    -- 예상 이동 시간 (분) — 평균 시속 30km 추정, 추후 카카오 API로 대체
    ROUND((ST_Distance(o.location, v_patient_location) / 1000 / 30 * 60)::NUMERIC, 0) AS travel_minutes,

    -- 서비스 적합성 점수 (0~100)
    ROUND(
      (ARRAY_LENGTH(
        ARRAY(SELECT UNNEST(v_needed_services) INTERSECT SELECT UNNEST(o.services)),
        1
      )::NUMERIC / GREATEST(ARRAY_LENGTH(v_needed_services, 1), 1) * 100)::NUMERIC,
      1
    ) AS service_match_score,

    -- 가용 인력 점수 (0~100)
    ROUND(
      GREATEST(0, (1 - (
        COALESCE((SELECT AVG(s.current_patient_count::NUMERIC / GREATEST(s.max_patients, 1))
                  FROM public.staff s WHERE s.org_id = o.id AND s.is_active), 0.5)
      )) * 100)::NUMERIC,
      1
    ) AS capacity_score,

    -- 평판 점수 (0~100)
    ROUND((o.rating_avg / 5 * 100)::NUMERIC, 1) AS reputation_score,

    -- 응답 속도 점수 (0~100)
    ROUND(GREATEST(0, (1 - o.response_avg_hours / 48) * 100)::NUMERIC, 1) AS response_score,

    -- 종합 점수 (가중 평균)
    ROUND((
      (1 - LEAST(ST_Distance(o.location, v_patient_location) / 1000 / p_radius_km, 1)) * 30  -- 거리 30%
      + ROUND(
          (ARRAY_LENGTH(
            ARRAY(SELECT UNNEST(v_needed_services) INTERSECT SELECT UNNEST(o.services)),
            1
          )::NUMERIC / GREATEST(ARRAY_LENGTH(v_needed_services, 1), 1) * 100)::NUMERIC,
          1
        ) * 0.25  -- 서비스 적합성 25%
      + GREATEST(0, (1 - (
          COALESCE((SELECT AVG(s.current_patient_count::NUMERIC / GREATEST(s.max_patients, 1))
                    FROM public.staff s WHERE s.org_id = o.id AND s.is_active), 0.5)
        )) * 100) * 0.15  -- 가용 인력 15%
      + (o.rating_avg / 5 * 100) * 0.20  -- 평판 20%
      + GREATEST(0, (1 - o.response_avg_hours / 48) * 100) * 0.10  -- 응답 속도 10%
    )::NUMERIC, 1) AS total_score

  FROM public.organizations o
  WHERE o.verification_status = 'verified'
    AND ST_DWithin(o.location, v_patient_location, p_radius_km * 1000)
    AND o.services && v_needed_services  -- 서비스 겹침
  ORDER BY total_score DESC
  LIMIT 5;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- =====================================================
-- RAG 하이브리드 검색 (벡터 + BM25 키워드)
-- =====================================================
CREATE OR REPLACE FUNCTION public.rag_hybrid_search(
  p_query_embedding vector(1536),
  p_query_text TEXT,
  p_limit INTEGER DEFAULT 5
)
RETURNS TABLE (
  id UUID,
  title TEXT,
  content TEXT,
  source TEXT,
  similarity NUMERIC,
  rank_score NUMERIC
) AS $$
BEGIN
  RETURN QUERY
  WITH vector_results AS (
    SELECT
      d.id,
      d.title,
      d.content,
      d.source,
      1 - (d.embedding <=> p_query_embedding) AS similarity
    FROM public.rag_documents d
    WHERE d.is_active = TRUE
    ORDER BY d.embedding <=> p_query_embedding
    LIMIT p_limit * 2
  ),
  keyword_results AS (
    SELECT
      d.id,
      d.title,
      d.content,
      d.source,
      ts_rank(
        to_tsvector('simple', d.content),
        plainto_tsquery('simple', p_query_text)
      ) AS keyword_rank
    FROM public.rag_documents d
    WHERE d.is_active = TRUE
      AND to_tsvector('simple', d.content) @@ plainto_tsquery('simple', p_query_text)
    LIMIT p_limit * 2
  )
  SELECT
    COALESCE(v.id, k.id) AS id,
    COALESCE(v.title, k.title) AS title,
    COALESCE(v.content, k.content) AS content,
    COALESCE(v.source, k.source) AS source,
    COALESCE(v.similarity, 0)::NUMERIC AS similarity,
    -- 하이브리드 점수: 벡터 유사도 70% + 키워드 30%
    (COALESCE(v.similarity, 0) * 0.7
     + COALESCE(k.keyword_rank, 0) * 0.3)::NUMERIC AS rank_score
  FROM vector_results v
  FULL OUTER JOIN keyword_results k ON v.id = k.id
  ORDER BY rank_score DESC
  LIMIT p_limit;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

### 6.2 레드플래그 감지 엔진 상세

```typescript
// supabase/functions/red-flag-detect/index.ts
// 의사 코드 (실제 구현 시 참조)

interface RedFlagRule {
  category: string;
  severity: 'yellow' | 'orange' | 'red';
  check: (current: VitalRecord, history: VitalRecord[]) => boolean;
  title: string;
  descriptionTemplate: string;
}

const RULES: RedFlagRule[] = [
  // ===== RED (긴급) =====
  {
    category: 'vital_critical',
    severity: 'red',
    check: (cur) => (cur.vitals.systolic_bp ?? 0) >= 180,
    title: '수축기 혈압 위험 수치',
    descriptionTemplate: '수축기 혈압 {value}mmHg — 즉시 의료진 확인 필요',
  },
  {
    category: 'vital_critical',
    severity: 'red',
    check: (cur) => (cur.vitals.systolic_bp ?? 999) <= 90,
    title: '수축기 혈압 저하',
    descriptionTemplate: '수축기 혈압 {value}mmHg — 저혈압 위험',
  },
  {
    category: 'fall',
    severity: 'red',
    check: (cur) => cur.performed_items.some(i => i.item === 'fall_report' && i.done),
    title: '낙상 발생',
    descriptionTemplate: '낙상이 보고되었습니다',
  },
  {
    category: 'consciousness',
    severity: 'red',
    check: (cur) => cur.consciousness === 'altered' || cur.consciousness === 'unresponsive',
    title: '의식 변화',
    descriptionTemplate: '의식 상태 변화 감지: {value}',
  },

  // ===== ORANGE (경고) =====
  {
    category: 'vital_trend',
    severity: 'orange',
    check: (cur, history) => {
      // 최근 5회 방문 중 혈압이 지속적으로 위험 수치에 접근
      const recentBPs = history.slice(-5).map(h => h.vitals.systolic_bp).filter(Boolean);
      return recentBPs.length >= 3 && recentBPs.every(bp => bp! >= 150);
    },
    title: '혈압 지속 상승',
    descriptionTemplate: '최근 방문에서 수축기 혈압이 지속적으로 150mmHg 이상',
  },
  {
    category: 'cognitive_change',
    severity: 'orange',
    check: (cur, history) => {
      const prev = history[history.length - 1];
      return prev && prev.consciousness === 'alert' && cur.consciousness === 'confused';
    },
    title: '인지 상태 변화',
    descriptionTemplate: '이전 방문 대비 인지 상태 저하 감지',
  },
  {
    category: 'weight_change',
    severity: 'orange',
    check: (cur, history) => {
      // 2주 내 3kg 이상 변화
      const twoWeeksAgo = history.filter(h =>
        daysBetween(h.created_at, cur.created_at) <= 14
      );
      if (twoWeeksAgo.length === 0) return false;
      const oldWeight = twoWeeksAgo[0].vitals.weight;
      const newWeight = cur.vitals.weight;
      if (!oldWeight || !newWeight) return false;
      return Math.abs(newWeight - oldWeight) >= 3;
    },
    title: '급격한 체중 변화',
    descriptionTemplate: '2주 내 체중 {value}kg 변화 감지',
  },

  // ===== YELLOW (주의) =====
  {
    category: 'vital_trend',
    severity: 'yellow',
    check: (cur, history) => {
      // 3회 연속 상승 추세
      const recent3 = [...history.slice(-2), cur].map(h => h.vitals.systolic_bp).filter(Boolean);
      return recent3.length === 3 && recent3[0]! < recent3[1]! && recent3[1]! < recent3[2]!;
    },
    title: '혈압 상승 추세',
    descriptionTemplate: '3회 연속 혈압 상승 추세 감지',
  },
  {
    category: 'medication_miss',
    severity: 'yellow',
    check: (cur, history) => {
      const recent2 = [...history.slice(-1), cur];
      return recent2.every(h =>
        h.performed_items.some(i => i.item === 'medication_check' && !i.done)
      );
    },
    title: '복약 연속 누락',
    descriptionTemplate: '2회 연속 복약 미확인',
  },
  {
    category: 'nutrition_decline',
    severity: 'yellow',
    check: (cur) => cur.nutrition_intake === 'poor' || cur.nutrition_intake === 'none',
    title: '식사량 감소',
    descriptionTemplate: '식사량 {value} — 영양 상태 주의',
  },
];

// 실행 플로우:
// 1. 체크아웃 시 visit_record INSERT 트리거 → Edge Function 호출
// 2. 해당 환자의 최근 10회 방문 기록 조회
// 3. 규칙 기반 체크 실행
// 4. AI 보조 분석 (Gemini): 규칙으로 감지 못한 패턴 추가 분석
// 5. 감지된 알림 → red_flag_alerts INSERT
// 6. severity별 알림 대상자에게 푸시/알림톡 발송
```

### 12.2 CI/CD 파이프라인

```yaml
# .github/workflows/ci.yml (PR 검증)
name: CI
on: [pull_request]
jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with: { node-version: 22 }
      - run: pnpm install --frozen-lockfile
      - run: pnpm turbo lint
      - run: pnpm turbo type-check
      - run: pnpm turbo test

# .github/workflows/deploy-web.yml (웹 배포 — Vercel이 자동 처리)
# → Vercel GitHub Integration으로 main 브랜치 push 시 자동 배포

# .github/workflows/deploy-mobile.yml (모바일 빌드 + 제출)
name: Mobile Deploy
on:
  push:
    tags: ['v*']  # 태그 푸시 시 빌드
jobs:
  build-and-submit:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        app: [patient-app, nurse-app]
    steps:
      - uses: actions/checkout@v4
      - uses: expo/expo-github-action@v8
        with:
          eas-version: latest
          token: ${{ secrets.EXPO_TOKEN }}
      - run: cd apps/${{ matrix.app }} && eas build --platform all --non-interactive
      - run: cd apps/${{ matrix.app }} && eas submit --platform all --non-interactive
```
